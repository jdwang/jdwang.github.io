apply和call

apply和call的相同点：改变作用域，this的指向；
不同点，传递的参数不同

[js]
function b(){
a.apply(this,arguments) //a.apply(this,["1","2"…])
//a.call(this,"1","2"…)
}
[/js]

apply和call使用的例子

[js]
function Cc(message){
this.message = message;
}
Cc.prototype = {
constructor:Cc,
showMessage:function(msg){
var msg = msg ||this.message
console.log(msg);
return msg
}
}
var ac = new Cc("hah");

function Bb(message){
    this.message = message;
}
Bb.prototype = {
    constructor:Bb,
    showing:function(){
        console.log(this)
        return ac.showMessage.apply(this,[&quot;fk&quot;])
    }
}
var bc = new Bb(&quot;who&quot;);
bc.showing();

[/js]

each函数的实现

[js]
function each(tagname,callback){
var lists = document.getElementsByTagName(tagname);
for(var i=0;i<list.length;i++){
callback.call(lists[i]);
}

}
each(&quot;li&quot;,function(){
       console.log(this);
})

[/js]

另一个例子

[js]
function Person(name,age,job){
this.name = name;
this.age = age;
this.job = job;
this.sayName = function(){
alert(this.name);
}
}
var jack = new Person("jack","18","f2e")
var obj = {
name:"won",
who:function(){
jack.sayName.apply(this);
}
}
obj.who()
[/js]
